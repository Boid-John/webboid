<script>
import {openURL} from 'quasar'
export default {
  data(){
    return {
      newPayoutAccount:"",
      page:0,
      pending:false,
      scatterAuth:null,
      scatterId:null
    }
  },
  mounted(){
    this.init()
  },
  methods:{
    openURL,
    reset(){
      this.page = 0
      this.newPayoutAccount = ""
    },
    init(){
      this.$root.$on('scatterAuth',(data)=>{
        this.scatterAuth = data
      })
      this.$root.$on('scatterId',(data)=>{
        this.scatterId = data
      })
      this.$root.$on('modal.updatePayout',(toggle) =>{
        if (toggle){
          this.thisModal.open()
        } else{
          this.thisModal.close()
        }
      })
      
    },
    async linkAccount(){
      const result = await this.$api.createAccountUpdateRequest({type:"EOS",newEos:this.scatterId.name})
      if(!result) alert('there was a problem with this request')
      if(result.success){
        this.page = 1
      }
    },
    modalOpened(){
      console.log('modal open')
      this.reset()
      window.olark('api.box.hide')
 
    },
    modalClosed(){
      console.log('modal closed')
      this.reset()
      window.olark('api.box.show')
 
    } 
  },
  computed:{
  },
  props:['thisModal','api','thisUser'],
  watch:{
    thisModal(modal){
      if (modal){
        this.thisModal.$on('close',this.modalClosed)
        this.thisModal.$on('open',this.modalOpened)
      }
    },
    thisUser(){
    }
  }
}
</script>
